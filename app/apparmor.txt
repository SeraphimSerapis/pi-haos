# Starting profile for the App container. This is intentionally conservative:
# the live Home Assistant configuration is readable but never writable, and
# private state is limited to the App's /data volume. The Pi child-process
# filesystem boundary is enforced by the Landlock launcher.
profile pi_homeassistant_agent flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  network inet stream,
  network inet6 stream,
  /app/** r,
  /opt/pi/** r,
  /app/bin/pi-sandbox ix,
  /usr/** r,
  /lib/** r,
  /lib64/** r,
  /etc/** r,
  /homeassistant/** r,
  /data/** rwk,
  /tmp/** rwk,
  deny /homeassistant/** wklx,
  deny /homeassistant/secrets.yaml rklx,
  deny /homeassistant/.storage/** rklx,
  deny /homeassistant/.cloud/** rklx,
  deny /run/docker.sock rwklx,
  deny /proc/** w,
  deny /sys/** w,
  deny /dev/** w,
}
